579  constexpr bool is_value = std::is_same_v<std::decay_t<T>, T>;
580

667 template<typename T>
668 auto& ModuleInput::change(T&& new_value) {
669  if constexpr(detail_::IsCString<T>::value) {
670  change_(wrap_value_(std::string(new_value)));
671  } else { // needed to avoid this branch being compiled for string literals
672 
673  constexpr bool is_value = std::is_same_v<T, std::decay_t<T>>;
674  constexpr bool is_rref = std::is_rvalue_reference_v<T>;
675 
676  type::any da_any;
677 
678  /* Fun time. The sde supports modules taking arguments in one of two
679  * ways:
680  * 1. By const reference (read-only)
681  * 2. By value
682  *
683  * When setting the actual value there's no stipulation requiring
684  * the user to provide us that value in the same form. Specifically
685  * they can provide us the value:
686  * 1. In read/write mode (by reference)
687  * 2. In read-only mode (by constant reference)
688  * 3. By value (constructed it in place)
689  *
690  * Of these three possible inputs, number 3 is the easiest. We
691  * basically just take ownership of it. For the other types of input
692  * what we do depends on how it's going to be used. If the value
693  * will be used in a read-only manner we just make a reference
694  * wrapper around it, otherwise we copy it.
695  */
696  if constexpr(is_value || is_rref) { // User gave us the input by value
697  da_any = wrap_value_(std::forward<T>(new_value));
698  m_is_actually_cref_ = false;
699  } else {
700  if(m_is_cref_) {
701  da_any = wrap_cref_(std::forward<T>(new_value));
702  m_is_actually_cref_ = true;
703  } else
704  da_any = wrap_value_(std::forward<T>(new_value));
705  }
706  change_(std::move(da_any));
707  }
708  return *this;
709 }
