146  shared_module at(const type::key& key) {
147  if(!count(key)) {
148  const std::string msg =
149  "ModuleManager has no module with key: '" + key + "'";
150  throw std::out_of_range(msg);
151  }
152  auto mod = m_modules.at(key);
153  // Loop over submodules filling them in from the defaults
154  for(auto& [k, v] : mod->submods()) {
155  const auto& type = v.type();
156  // Only change non-ready submodules
157  if(!v.ready() && m_defaults.count(type)) {
158  // Recursive to make sure that that module gets filled in
159  auto default_mod = at(m_defaults.at(type));
160  // Only change if the module is also ready
161  if(default_mod->ready(m_inputs.at(type)))
162  mod->change_submod(k, default_mod);
163  }
164  }
165  return mod;
166  }
