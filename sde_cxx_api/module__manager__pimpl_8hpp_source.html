115  shared_module at(const type::key& key) {
116  auto mod = m_modules.at(key);
117  // Loop over submodules filling them in from the defaults
118  for(auto& [k, v] : mod->submods()) {
119  const auto& type = v.type();
120  // Only change non-ready submodules
121  if(!v.ready() && m_defaults.count(type)) {
122  // Recursive to make sure that that module gets filled in
123  auto default_mod = at(m_defaults.at(type));
124  // Only change if the module is also ready
125  if(default_mod->ready(m_inputs.at(type)))
126  mod->change_submod(k, default_mod);
127  }
128  }
129  return mod;
130  }
