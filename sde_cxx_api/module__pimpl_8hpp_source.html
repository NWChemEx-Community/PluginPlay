767 inline auto ModulePIMPL::run(type::input_map ps) {
768  auto time_now = time_stamp();
769  m_timer_.reset();
770  assert_mod_();
771  // Check the inputs we were just given
772  for(const auto& [k, v] : ps)
773  if(!v.ready()) throw std::runtime_error("Inputs are not ready");
774 
775  // Merge with bound and see if we are ready
776  if(!ready(ps)) {
777  // Make a dummy module with this PIMPL so we can print out why it's not
778  // ready.
779  Module dummy(std::make_unique<ModulePIMPL>(*this));
780  throw std::runtime_error(print_not_ready(dummy, ps));
781  }
782 
783  lock();
784 
785  ps = merge_inputs_(ps);
786  // Check cache
787  auto hv = get_hash_(ps);
788 
789  if(is_memoizable() && m_cache_ && m_cache_->count(hv)) {
790  m_timer_.record(time_now);
791  return m_cache_->at(hv);
792  }
793 
794  // not there so run
795  auto rv = m_base_->run(std::move(ps), m_submods_);
796  if(!m_cache_ || !is_memoizable()) {
797  m_timer_.record(time_now);
798  return rv;
799  }
800 
801  // cache result
802  m_cache_->emplace(hv, std::move(rv));
803  m_timer_.record(time_now);
804  return m_cache_->at(hv);
805 }
