683 inline auto ModulePIMPL::run(type::input_map ps) {
684  auto time_now = time_stamp();
685  m_timer_.reset();
686  assert_mod_();
687  // Check the inputs we were just given
688  for(const auto& [k, v] : ps)
689  if(!v.ready()) throw std::runtime_error("Inputs are not ready");
690 
691  // Merge with bound and see if we are ready
692  if(!ready(ps)) throw std::runtime_error("Module is not ready to be run");
693 
694  lock();
695 
696  ps = merge_inputs_(ps);
697  // Check cache
698  auto hv = get_hash_(ps);
699  if(false && m_cache_ && m_cache_->count(hv)) {
700  m_timer_.record(time_now);
701  return m_cache_->at(hv);
702  }
703 
704  // not there so run
705  auto rv = m_base_->run(std::move(ps), m_submods_);
706  if(true || !m_cache_) {
707  m_timer_.record(time_now);
708  return rv;
709  }
710 
711  // cache result
712  m_cache_->emplace(hv, std::move(rv));
713  m_timer_.record(time_now);
714  return m_cache_->at(hv);
715 }
