288 void PROP_TYPE::wrap_guts_(T&& rv, U&& builder, V&& value, Args&&... args) {
289  using traits_type = typename std::decay_t<U>::traits_type;
290  using tuple_of_fields = typename traits_type::tuple_of_fields;
291  using type = std::tuple_element_t<ArgI, tuple_of_fields>;
292  constexpr bool is_ref = std::is_reference_v<type>;
293  static_assert(std::is_convertible_v<V, type>,
294  "Wrap argument is of incorrect type.");
295 
296  auto key = (builder.begin() + ArgI)->first;
297  if constexpr(is_ref) {
298  rv.at(key).change(static_cast<type>(value));
299  } else {
300  rv.at(key).change(std::forward<V>(value));
301  }
302  if constexpr(sizeof...(Args) > 0)
303  wrap_guts_<ArgI + 1>(std::forward<T>(rv), std::forward<U>(builder),
304  std::forward<Args>(args)...);

325 auto PROP_TYPE::unwrap_guts_(T&& builder, U&& rv) {
326  using tuple_of_fields = typename T::traits_type::tuple_of_fields;
327  constexpr auto nargs = std::tuple_size_v<tuple_of_fields>;
328  if constexpr(ArgI == nargs)
329  return std::make_tuple();
330  else {
331  using type = std::tuple_element_t<ArgI, tuple_of_fields>;
332  auto key = (builder.begin() + ArgI)->first;
333  auto lhs = std::tuple<type>(rv.at(key).template value<type>());
334  auto rhs =
335  unwrap_guts_<ArgI + 1>(std::forward<T>(builder), std::forward<U>(rv));
336  return std::tuple_cat(std::move(lhs), std::move(rhs));
