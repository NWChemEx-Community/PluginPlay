608 inline bool ModuleBase::operator==(const ModuleBase& rhs) const noexcept {
609  if(type() != rhs.type()) return false;
610  if(has_description() != rhs.has_description()) return false;
611  if(has_description() && get_desc() != rhs.get_desc()) return false;
612  if(m_citations_ != rhs.m_citations_) return false;
613  return std::tie(inputs(), results(), submods(), property_types()) ==
614  std::tie(rhs.inputs(), rhs.results(), rhs.submods(),
615  rhs.property_types());
