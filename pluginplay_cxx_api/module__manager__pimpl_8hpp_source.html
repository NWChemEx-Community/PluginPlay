121  void add_module(type::key key, module_base_ptr base) {
122  assert_unique_key_(key);
123  auto uuid = utility::generate_uuid();
124  auto internal_cache = m_caches.get_or_make_user_cache(uuid);
125  base->set_cache(internal_cache);
126  base->set_runtime(m_runtime_);
127  base->set_uuid(uuid);
128  auto module_cache = m_caches.get_or_make_module_cache(key);
129  std::unique_ptr<ModulePIMPL> pimpl;
130  if(base->is_python()) {
131  // This is a hacky patch to allow multiple python modules to be
132  // added while avoiding the type_index collisions.
133  // TODO: remove when a more elegant solution is determined
134  m_py_bases[key] = base;
135  pimpl =
136  std::make_unique<ModulePIMPL>(m_py_bases[key], module_cache);
137  } else {
138  std::type_index type(base->type());
139  if(!m_bases.count(type)) m_bases[type] = base;
140  pimpl = std::make_unique<ModulePIMPL>(m_bases[type], module_cache);
141  }
142  auto ptr = std::make_shared<Module>(std::move(pimpl));
143  ptr->set_name(key);
144  m_modules.emplace(std::move(key), ptr);
145  }

182  shared_module at(const type::key& key) {
183  if(!count(key)) {
184  const std::string msg =
185  "ModuleManager has no module with key: '" + key + "'";
186  throw std::out_of_range(msg);
187  }
188  auto mod = m_modules.at(key);
189  // Loop over submodules filling them in from the defaults
190  for(auto& [k, v] : mod->submods()) {
191  const auto& type = v.type();
192  // Only change non-ready submodules
193  if(!v.ready() && m_defaults.count(type)) {
194  // Recursive to make sure that that module gets filled in
195  auto default_mod = at(m_defaults.at(type));
196  // Only change if the module is also ready
197  if(default_mod->ready(m_inputs.at(type)))
198  mod->change_submod(k, default_mod);
199  }
200  }
201  return mod;
202  }

210  bool operator==(const ModuleManagerPIMPL& rhs) const {
211  // Try to get out early
212  if(m_bases.size() != rhs.m_bases.size()) return false;
213  if(m_modules.size() != rhs.m_modules.size()) return false;
214  if(m_defaults.size() != rhs.m_defaults.size()) return false;
215 
216  // TODO: Remove with the rest of the python hack
217  if(m_py_bases.size() != rhs.m_py_bases.size()) return false;
218  for(const auto& [k, v] : rhs.m_py_bases) {
219  if(!m_py_bases.count(k)) return false;
220  if(*m_py_bases.at(k) != *v) return false;
221  }
222 
223  // Skip checking the values b/c implementations are compared by type
224  for(const auto& [k, v] : rhs.m_bases) {
225  if(!m_bases.count(k)) return false;
226  }
227 
228  // Need to check the values b/c user may have switched options
229  for(const auto& [k, v] : rhs.m_modules) {
230  if(!m_modules.count(k)) return false;
231  if(*m_modules.at(k) != *v) return false;
232  }
