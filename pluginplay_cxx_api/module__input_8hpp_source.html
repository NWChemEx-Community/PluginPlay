579 bool ModuleInput::is_valid(T&& new_value) const {
580  constexpr bool is_value = std::is_same_v<std::decay_t<T>, T>;
581 
582  // If we got a value we can't directly forward it as it's basically a
583  // temporary, this first case creates an object to avoid this.

593 T ModuleInput::value() {
594  // Did the user ask for the value back by copy?
595  constexpr bool by_value = std::is_same_v<std::decay_t<T>, T>;

597  // Want a const reference or a copy? Defer to ModuleInput::value() const

607  return detail_::AnyCast<T>(get_());

668 template<typename T>
669 auto& ModuleInput::change(T&& new_value) {
670  if constexpr(detail_::IsCString<T>::value) {
671  change_(wrap_value_(std::string(new_value)));
672  } else { // needed to avoid this branch being compiled for string literals
673 
674  constexpr bool is_value = std::is_same_v<T, std::decay_t<T>>;
675  constexpr bool is_rref = std::is_rvalue_reference_v<T>;
676 
677  type::any da_any;
678 
679  /* Fun time. The pluginplay supports modules taking arguments in one of
680  * two ways:
681  * 1. By const reference (read-only)
682  * 2. By value
683  *
684  * When setting the actual value there's no stipulation requiring
685  * the user to provide us that value in the same form. Specifically
686  * they can provide us the value:
687  * 1. In read/write mode (by reference)
688  * 2. In read-only mode (by constant reference)
689  * 3. By value (constructed it in place)
690  *
691  * Of these three possible inputs, number 3 is the easiest. We
692  * basically just take ownership of it. For the other types of input
693  * what we do depends on how it's going to be used. If the value
694  * will be used in a read-only manner we just make a reference
695  * wrapper around it, otherwise we copy it.
696  */
697  if constexpr(is_value || is_rref) { // User gave us the input by value
698  da_any = wrap_value_(std::forward<T>(new_value));
699  m_is_actually_cref_ = false;
700  } else {
701  if(m_is_cref_) {
702  da_any = wrap_cref_(std::forward<T>(new_value));
703  m_is_actually_cref_ = true;
704  } else
705  da_any = wrap_value_(std::forward<T>(new_value));
706  }
707  change_(std::move(da_any));
708  }
709  return *this;
710 }
