736 template<typename property_type, typename... Args>
737 auto Module::run_as(Args&&... args) {
738  check_property_type_(type::rtti{typeid(property_type)});
739  auto temp = inputs();
740  temp = property_type::wrap_inputs(temp, std::forward<Args>(args)...);
741  using r_type = decltype(property_type::unwrap_results(run(temp)));
742  using clean_t = std::decay_t<r_type>;
743  if constexpr(std::is_same_v<clean_t, void>) {
744  property_type::unwrap_results(run(temp));
745  } else {
746  auto rv = property_type::unwrap_results(run(temp));
747 
748  if constexpr(std::tuple_size_v<clean_t> == 1) {
749  return std::get<0>(rv);
750  } else {
751  return rv;
752  }
753  }
754 }
