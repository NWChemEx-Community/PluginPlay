288 template<typename T, typename Op>
289 std::string CompareTo<T, Op>::str() const {
290  std::string rv;
291 
292  constexpr bool is_not_equal = std::is_same_v<Op, std::not_equal_to<T>>;
293  constexpr bool is_greater = std::is_same_v<Op, std::greater<T>>;
294  constexpr bool is_less = std::is_same_v<Op, std::less<T>>;
295  constexpr bool is_greater_equal = std::is_same_v<Op, std::greater_equal<T>>;
296  constexpr bool is_less_equal = std::is_same_v<Op, std::less_equal<T>>;
297 
298  if constexpr(is_not_equal) {
299  rv += "!=";
300  } else if constexpr(is_greater) {
301  rv += ">";
302  } else if constexpr(is_less) {
303  rv += "<";
304  } else if constexpr(is_greater_equal) {
305  rv += ">=";
306  } else if constexpr(is_less_equal) {
307  rv += "<=";
308  } else {
309  rv += "<undefined operator> ";
310  }
311 
312  return rv + " " + std::to_string(m_value);
313 }
