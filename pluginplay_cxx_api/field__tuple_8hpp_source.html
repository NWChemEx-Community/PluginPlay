314 template<typename LHSType, typename RHSType, std::size_t depth = 0>
315 auto union_guts(LHSType&& lhs, const RHSType& rhs) {
316  if constexpr(rhs.size() == depth) { // End recursion
317  return lhs;
318  } else { // Non-empty RHS
319  auto itr = rhs.begin();
320  std::advance(itr, depth);
321  const auto& [k, v] = *itr;
322  using tuple_type = typename RHSType::traits_type::tuple_of_fields;
323  using v_type = std::tuple_element_t<depth, tuple_type>;
324  auto rv = lhs.template add_field<v_type>(k);
325  if constexpr(std::decay_t<LHSType>::is_input) {
326  if(v.has_value()) { rv.set_default(v.template value<v_type>()); }
327  }
328  return union_guts<decltype(rv), RHSType, depth + 1>(std::move(rv), rhs);
329  }
330 }
