    list(
       APPEND SRCS PyBindings/ExportSDE.cpp
                   PyBindings/PyParameters.cpp

                   PyBindings/PyModuleManager.cpp
    )

       APPEND INCS PyBindings/PyModule.hpp

    )

        sde
        SOURCES ${SRCS}

    )

        set_target_properties(sde PROPERTIES SUFFIX .so)
